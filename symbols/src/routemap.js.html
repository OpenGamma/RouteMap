110         /**
111          * builds the internal representation of a rule based on the route definition
112          * @inner
113          * @name RouteMap#compile
114          * @function
115          * @param {String} route
116          * @throws {SyntaxError} if any portion of a rule definition follows a <code>*</code> directive
117          * @throws {SyntaxError} if a required scalar follows an optional scalar
118          * @throws {SyntaxError} if a rule cannot be parsed
119          * @type {Object}
120          * @returns {Object} a compiled object, for example, the rule <code>'/foo/:id/type:?/rest:*'</code> would return
121          * an object of the form: <blockquote><pre>{
122          *     page:'/foo',
123          *     rules:{
124          *         keyvals:[{name: 'type', required: false}],
125          *         scalars:[{name: 'id', required: true}],
126          *         star:'rest' // false if not defined
127          *     }
128          * }
129          * @see RouteMap.add
130          * @see RouteMap.hash
131          * @see RouteMap.remove
132          */
133         compile = (function () {
134             var memo = {}; // compile is slow so cache compiled objects here
135             return function (route) {
136                 var self = 'compile', param, compiled_route, orig = route, index;
137                 route = route[0] === SL ? route : ~(index = route.indexOf(SL)) ? route.slice(index) : null;
138                 if (!route) throw new SyntaxError(self + ': the route "' + orig + '" was not understood');
139                 if (memo[route]) return memo[route];
140                 compiled_route = route.split(SL).reduce(function (acc, val) {
141                     var rules = acc.rules, scalars = rules.scalars, keyvals = rules.keyvals;
142                     if (rules.star) throw new SyntaxError(self + ': no rules can follow a * directive');
143                     // construct the name of the page
144                     if (!~val.search(token_exp) && !scalars.length && !keyvals.length) return acc.page.push(val), acc;
145                     // construct the parameters
146                     if (val.match(star_exp)) return (rules.star = RegExp.$2 || RegExp.$3), acc;
147                     if (val.match(scalar_exp)) {
148                         if (acc.has_optional_scalar) // no scalars can follow optional scalars
149                             throw new SyntaxError(self + ': "' + val + '" cannot follow an optional rule');
150                         if (!!RegExp.$2) acc.has_optional_scalar = val;
151                         return scalars.push({name: RegExp.$1, required: !RegExp.$2}), acc;
152                     }
153                     if (val.match(keyval_exp)) return keyvals.push({name: RegExp.$1, required: !RegExp.$2}), acc;
154                     throw new SyntaxError(self + ': the rule "' + val + '" was not understood');
155                 }, {page: [], rules: {scalars: [], keyvals: [], star: false}, has_optional_scalar: ''});
156                 delete compiled_route.has_optional_scalar; // this is just a temporary value and should not be exposed
157                 compiled_route.page = compiled_route.page.join(SL).replace(new RegExp(SL + '$'), '') || SL;
158                 return memo[route] = compiled_route;
159             };

169          * keyvals, and stars; scalars are individual values in a URL (all URL values are separate by the

176          * <code>/foo/23/45/attr=something/subattr=something_else</code></blockquote>

178          * <code><pre>{
179          *      id:'23',
180          *      subid:'45',
